IF NOT EXISTS (SELECT * FROM sys.tables WHERE name = 'Employee' AND type = 'U')
BEGIN
    CREATE TABLE Employee (
        Id INT IDENTITY(1,1) PRIMARY KEY,
        Name NVARCHAR(100) NOT NULL,
        Age INT,
        Department NVARCHAR(50),
        Salary DECIMAL(10,2)
    );
    
    -- Insert sample data
    INSERT INTO Employee (Name, Age, Department, Salary)
    VALUES ('John Doe', 30, 'IT', 60000),
           ('Jane Smith', 28, 'HR', 50000),
           ('Michael Brown', 35, 'Finance', 70000);
END;

IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND name = 'GetEmployee')
BEGIN
    DROP PROCEDURE GetEmployee;
END;
GO

CREATE PROCEDURE GetEmployee
AS
BEGIN
    SET NOCOUNT ON;
    SELECT * FROM Employee;
END;
GO
